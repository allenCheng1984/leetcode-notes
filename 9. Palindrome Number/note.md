int main() {
    int x = 214747412;
    // int x = 53262341;
    int i = 0, j = x;

    printf("i = %d, x = %d\n", i, x);
    while(x){
        i = i*10 + x%10;
        x /= 10;

        printf("i = %d, x = %d\n", i, x);
    }
    printf("i = %d, j = %d", i, j);
}
